	\begin{problem}{A}
		\part{} Assume $x = y$.

		\part{1.)} if $y = 2$, find $x$.
		\part{b} if $x = -3$, find $y$.
		\part{(z)} Given:
			\begin{equation}
				x = \sum_{n=0}^{\infty}{\left(\frac{1}{2}\right)^n}
			\end{equation}
		find $y$.

		\solution
		\part{1.)} $x = 2$.
		\part{b} $y = -3$.
		\part{(z)}
			\begin{equation}
				y = x = \sum_{n=0}^{\infty} \left(\frac{1} {2}\right)^n
				  = \frac {1} {1 - \frac{1} {2}} = 2
			\end{equation}
	\end{problem}
